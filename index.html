        let currentResult = { script: '', hooks: [], hashtags: [] };


        function switchTab(tabName) {

            // Hide all tabs

            document.querySelectorAll('.tab-content').forEach(tab => {

                tab.classList.add('hidden');

            });

            document.querySelectorAll('.tab-button').forEach(btn => {

                btn.classList.remove('active');

            });


            // Show selected tab

            document.getElementById(`${tabName}-tab`).classList.remove('hidden');

            event.target.classList.add('active');

        }


        async function generateScript() {

            const network = document.getElementById('network').value;

            const language = document.getElementById('language').value;

            const niche = document.getElementById('niche').value;

            const angle = document.getElementById('angle').value;

            const customPrompt = document.getElementById('customPrompt').value;


            if (!niche || !angle) {

                showError('Veuillez remplir la Niche et l\'Angle');

                return;

            }


            showLoading();

            hideError();


            try {

                // API Call - À adapter avec votre endpoint

                const response = await fetch('YOUR_API_ENDPOINT_HERE', {

                    method: 'POST',

                    headers: {

                        'Content-Type': 'application/json',

                        // 'Authorization': 'Bearer YOUR_API_KEY' - À ajouter

                    },

                    body: JSON.stringify({

                        network,

                        language,

                        niche,

                        angle,

                        customPrompt

                    })

                });


                if (!response.ok) {

                    if (response.status === 401) {

                        showError('Connecte-toi ou configure ton API key pour générer un script.');

                    } else {

                        showError('Erreur lors de la génération. Réessaye.');

                    }

                    hideLoading();

                    return;

                }


                const data = await response.json();

                

                // Parse results

                currentResult = {

                    script: data.script || '',

                    hooks: data.hooks || [],

                    hashtags: data.hashtags || []

                };


                displayResults();

                hideLoading();

                document.getElementById('copyBtn').disabled = false;


            } catch (error) {

                console.error('Error:', error);

                showError('Erreur réseau. Vérifie ta connexion ou l\'API.');

                hideLoading();

            }

        }


        function displayResults() {

            // Display Script

            document.getElementById('scriptResult').textContent = currentResult.script;


            // Display Hooks

            const hooksHtml = currentResult.hooks

                .map(hook => `<div class="bg-black/30 p-3 rounded-lg border-l-4 border-pink-500">${hook}</div>`)

                .join('');

            document.getElementById('hooksResult').innerHTML = hooksHtml || '<p class="text-gray-400">Aucun hook généré</p>';


            // Display Hashtags

            const hashtagsHtml = currentResult.hashtags

                .map(tag => `<span class="bg-purple-500/20 px-3 py-1 rounded-full text-sm">${tag}</span>`)

                        showError('Connecte-toi ou configure ton API key pour générer un script.');

                    } else {

                        showError('Erreur lors de la génération. Réessaye.');

                    }

                    hideLoading();

                    return;

                }


                const data = await response.json();

                

                // Parse results

                currentResult = {

                    script: data.script || '',

                    hooks: data.hooks || [],

                    hashtags: data.hashtags || []

                };


                displayResults();

                hideLoading();

                document.getElementById('copyBtn').disabled = false;


            } catch (error) {

                console.error('Error:', error);

                showError('Erreur réseau. Vérifie ta connexion ou l\'API.');

                hideLoading();

            }

        }


        function displayResults() {

            // Display Script

            document.getElementById('scriptResult').textContent = currentResult.script;


            // Display Hooks

            const hooksHtml = currentResult.hooks

                .map(hook => `<div class="bg-black/30 p-3 rounded-lg border-l-4 border-pink-500">${hook}</div>`)

                .join('');

            document.getElementById('hooksResult').innerHTML = hooksHtml || '<p class="text-gray-400">Aucun hook généré</p>';


            // Display Hashtags

            const hashtagsHtml = currentResult.hashtags

                .map(tag => `<span class="bg-purple-500/20 px-3 py-1 rounded-full text-sm">${tag}</span>`)

                .join('');

            document.getElementById('hashtagsResult').innerHTML = hashtagsHtml || '<p class="text-gray-400">Aucun hashtag généré</p>';


            document.getElementById('resultContainer').classList.remove('hidden');

        }


        function copyToClipboard() {

            const text = `${currentResult.script}\n\n${currentResult.hooks.join('\n')}\n\n${currentResult.hashtags.join(' ')}`;

            navigator.clipboard.writeText(text).then(() => {

                const btn = document.getElementById('copyBtn');

                const originalText = btn.textContent;

                btn.textContent = '✅ Copié!';

                setTimeout(() => {

                    btn.textContent = originalText;

                }, 2000);

            });

        }


        function showLoading() {

            document.getElementById('loadingContainer').classList.remove('hidden');

            document.getElementById('resultContainer').classList.add('hidden');

        }


        function hideLoading() {

            document.getElementById('loadingContainer').classList.add('hidden');

        }


        function showError(message) {

            document.getElementById('errorMessage').textContent = message;

            document.getElementById('errorContainer').classList.remove('hidden');

            document.getElementById('resultContainer').classList.add('hidden');

        }


        function hideError() {

            document.getElementById('errorContainer').classList.add('hidden');

        }


        // Event Listeners

        document.getElementById('generateBtn').addEventListener('click', generateScript);

        document.getElementById('copyBtn').addEventListener('click', copyToClipboard);


        // Enter key to generate

        document.getElementById('customPrompt').addEventListener('keydown', (e) => {

            if (e.key === 'Enter' && e.ctrlKey) generateScript();

        });
